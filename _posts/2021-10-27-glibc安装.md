# gblic安装

```
git clone https://sourceware.org/git/glibc.git
cd glibc
# 选择你的stable版本
git checkout release/2.23/master

# 有时候会说gawk太旧
sudo apt-get install gawk
sudo apt-get install bison

mkdir glibc-build
cd glibc-build
../glibc-2.23/configure -prefix=/some/path
make
make install


```
